بردار مقادیر مشاهده شده و بردار میانگین و ماتریس کوواریانس موجود در سلول

``` r
f <- function(x,mu,siggma){
  
  densitty <- (1/(sqrt(((2*pi)^3))))*exp((-1/2)*(t(x-mu)%*%solve(siggma)%*%(x-mu)))
  print(densitty)
}

x <- c(1,0.5,-1)
mu <- c(0.25,-0.25,0)
siggma <- matrix(c(9,0,-1,0,7,2,-1,2,5),3)
f(x,mu,siggma)
```
